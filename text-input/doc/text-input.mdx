# TextInput

A component to allow users to input text.

## Usage

```tsx
import * as React from 'react';
import { TextInput } from '@divriots/starter-origami/text-input';

const MyComponent = () => {
  const [text, setText] = React.useState('');

  return (
    <TextInput
      label="Email"
      value={text}
      onChangeText={(text) => setText(text)}
    />
  );
};
```
